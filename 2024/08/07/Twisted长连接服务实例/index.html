背景

最近工作的需求，使用Python实现TCP长连接服务，给Java Netty的客户端调用，同时Python的Web服务还需要对Java的TCP客户端发送数据。

这里使用Twisted实现一个长连接服务，以及一个HTTP服务，HTTP服务作为通道，连接Django的后端工程，通过管道发数据给Java的客户端。

如果Java客户端发送了数据给twisted，twisted服务还有处理业务逻辑，这部分需要使用后端的异步线程去处理。

实例代码

from twisted.internet import protocol, reactor, threads
from twisted.web import server, resource
from twisted.internet.task import LoopingCall
from twisted.python.threadpool import ThreadPool
from threading import Lock
import time

def slow_function(param):
    """
    这是一个模拟耗时任务的函数。
    """
    print(f"Starting slow function with param: {param}")
    time.sleep(5)  # 模拟耗时操作
    print("Slow function finished")
    return f"Processed {param}"

def on_success(arg):
    print(arg)


def on_failure(arg):
    print(arg)


class HeartbeatProtocol(protocol.Protocol):
    def __init__(self):
        self.last_heartbeat = None
        self.peer_ip = None

    def connectionMade(self):
        self.peer_ip = self.transport.getPeer().host
        self.factory.add_connection(self)
        self.last_heartbeat = reactor.seconds()
        print(f"Connection made: {self.peer_ip}")

        param = 'message to function'
        d = threads.deferToThread(slow_function, param)
        d.addCallbacks(on_success, on_failure)


    def connectionLost(self, reason):
        self.factory.remove_connection(self)
        print(f"Connection lost: {self.peer_ip}. Reason: {reason.getErrorMessage()}")

    def dataReceived(self, data):
        if data.strip() == b'heartbeat':
            self.last_heartbeat = reactor.seconds()
            self.transport.write(b'heartbeat acknowledged\n')
        else:
            print(f"Received data: {data.decode()}")

    def send_message(self, message):
        self.transport.write(message.encode() + b'\n')

class HeartbeatFactory(protocol.Factory):
    def __init__(self):
        self.connections = []
        self.connection_map = {}  # Mapping from IP address to protocol instance
        self.heartbeat_timeout = 10  # seconds
        self.lock = Lock()  # Create a lock for thread safety

    def buildProtocol(self, addr):
        protocol = HeartbeatProtocol()
        protocol.factory = self
        return protocol

    def check_heartbeats(self):
        def worker():
            with self.lock:  # Ensure exclusive access to connections and connection_map
                print('CHECK')
                current_time = reactor.seconds()
                for conn in self.connections[:]:
                    if current_time - conn.last_heartbeat > self.heartbeat_timeout:
                        print(f"Connection timed out: {conn.peer_ip}")
                        conn.transport.loseConnection()
        #threads.deferToThread(worker)
        d = threads.deferToThread(worker)
        d.addErrback(lambda failure: print(f"Error in heartbeat check: {failure.getErrorMessage()}"))

    def send_to_ip(self, ip, message):
        print(f"Trying to send message to IP: {ip}")
        with self.lock:  # Ensure exclusive access to connection_map
            if ip in self.connection_map:
                self.connection_map[ip].send_message(message)
            else:
                print(f"IP {ip} not found in connection map.")

    def add_connection(self, protocol):
        if protocol.peer_ip is not None:
            with self.lock:  # Ensure exclusive access to connections and connection_map
                self.connections.append(protocol)
                self.connection_map[protocol.peer_ip] = protocol
                print(f"Connection added: {protocol.peer_ip}")

    def remove_connection(self, protocol):
        if protocol.peer_ip in self.connection_map:
            with self.lock:  # Ensure exclusive access to connections and connection_map
                self.connections.remove(protocol)
                del self.connection_map[protocol.peer_ip]
                print(f"Connection removed: {protocol.peer_ip}")

class MessageResource(resource.Resource):
    isLeaf = True

    def __init__(self, factory):
        self.factory = factory

    def render_POST(self, request):
        ip = request.args.get(b'ip', [b''])[0].decode()
        message = request.args.get(b'message', [b''])[0].decode()
        print('Request received:', ip, message)
        if ip:
            self.factory.send_to_ip(ip, message)
            return b"Message sent to specific IP."
        else:
            return b"IP address not provided."

if __name__ == '__main__':
    # Configure the thread pool
    thread_pool = ThreadPool(minthreads=4, maxthreads=16)
    thread_pool.start()
    reactor.threadpool = thread_pool

    factory = HeartbeatFactory()
    reactor.listenTCP(12345, factory)  # TCP service for Java clients

    # HTTP service to receive messages from Django
    site = server.Site(MessageResource(factory))
    reactor.listenTCP(12346, site)  # HTTP service on port 12346

    heartbeat_check = LoopingCall(factory.check_heartbeats)
    heartbeat_check.start(1)  # Check every 1 second

    print("Twisted server started.")
    reactor.run()

总结

twisted实现长连接很方便，提供了很多内置功能，比如心跳检查等功能，只要在对应的回调函数实现具体的业务逻辑即可。
