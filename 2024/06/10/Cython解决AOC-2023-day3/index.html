背景

本文主要记录解决AOC-2023-day3的问题过程，主要使用Python和Cython分别测试效果。

题目大意

第一部分

给出一个矩阵：

467..114..
...*......
..35..633.
......#...
617*......
.....+.58.
..592.....
......755.
...$.*....
.664.598..

提取其中所有的数字，连续数字记为一个数字，需要把其中被.包围的数字去除掉，比如114和58，因为它们不管从各个方位看，都只被.包围。最后把没有被.包围的数字加和。

思路有点像Life Game，对每个数字，判断它周围的所有点是否全是.全部是.的不要。其他的数字加和。

Python实现：

import time

def read_data(s):
    """读取数字到列表"""
    s_list = []
    for line in s.strip().split('\n'):
        s_list.append(line)
    return s_list


def get_num_or_zero(s_list, length_row, length_column, index_i, index_j_start, index_j_end):
    """获取数值或者0，不满足条件返回0，否则返回数字"""

    other_ele = [(index_i, index_j_start-1), (index_i, index_j_end+1)]

    for j in range(index_j_start-1, index_j_end+2):
        other_ele.append((index_i-1, j))
        other_ele.append((index_i+1, j))

    ret = []

    for ele in other_ele:
        if ele[0] < 0 or ele[1] >= length_column or ele[1] < 0 or ele[0] >= length_row:
            continue
        else:
            ret.append(ele)
    for ele in ret:
        if s_list[ele[0]][ele[1]] != '.':
            return int(s_list[index_i][index_j_start: index_j_end+1])
    return 0


def get_all_sum(s_list):
    """获取所有的数字"""
    length_row = len(s_list)
    length_column = len(s_list[0])
    all_sum = 0
    for i in range(length_row):
        line_count = 0
        line_sum = 0
        start_j = None
        end_j = None
        for j in range(length_column):
            if s_list[i][j] in '0123456789' and start_j is None:
                start_j = j
                end_j = j
            elif s_list[i][j] in '0123456789' and start_j is not None:
                end_j = j
            elif s_list[i][j] not in '0123456789' and start_j is not None and end_j is not None:
                num = get_num_or_zero(s_list, length_row, length_column, i, start_j, end_j)
                line_sum += num
                all_sum += num
                # print(f'line {i}, num {num}')
                line_count += 1
                start_j = None
                end_j = None

        # 最后一个数字也要处理
        if start_j is not None and end_j is not None:
            num = get_num_or_zero(s_list, length_row, length_column, i, start_j, end_j)
            line_sum += num
            all_sum += num
            # print(f'line {i}, num {num}')
            line_count += 1
        # print(f'line {i}, count {line_count}, sum {line_sum}')
    # print(f'Total sum: {all_sum}')
    return all_sum


# 使用给定的字符串
s = '''
467..114..
...*......
..35..633.
......#...
617*......
.....+.58.
..592.....
......755.

.664.598..

all_list = read_data(s)

第二部分

缺失的零件并不是唯一的问题——发动机的一个齿轮出了问题。齿轮是与两个零件号正好相邻的任何符号。

它的传动比是这两个数字相乘的结果。

这一次，你需要找到每个齿轮的传动比，并将其全部相加，以便工程师能够找出需要更换的齿轮。
再次考虑相同的发动机示意图：

467..114..
...*......
..35..633.
......#...
617*......
.....+.58.
..592.....
......755.
...$.*....
.664.598..

在这个示意图中，有两个齿轮。第一个在左上角；它的零件号分别为467和35，因此其传动比为16345。第二档在右下角；其传动比为451490。（与617相邻的*不是齿轮，因为它只与一个零件号相邻。）将所有传动比相加可得到467835。

这里是找到所有*周围的2个数字，没有两个数字可以按0处理，计算乘积，把所有乘积加起来。

容易想到的算法，可以基于第一部分的代码：

对每个数字，返回它周围的*索引，如果有两个数字的索引一样，那么把他们两个乘起来，把所有结果加和即可。

Python实现如下：

import time

def read_data(s):
    """读取数字到列表"""
    s_list = []
    for line in s.strip().split('\n'):
        s_list.append(line)
    return s_list


def get_num_or_zero(s_list, length_row, length_column, index_i, index_j_start, index_j_end):
    """获取数值或者0，不满足条件返回0，否则返回数字"""

    other_ele = [(index_i, index_j_start-1), (index_i, index_j_end+1)]

    for j in range(index_j_start-1, index_j_end+2):
        other_ele.append((index_i-1, j))
        other_ele.append((index_i+1, j))

    ret = []
    res = {}

    for ele in other_ele:
        if ele[0] < 0 or ele[1] >= length_column or ele[1] < 0 or ele[0] >= length_row:
            continue
        else:
            ret.append(ele)
    for ele in ret:
        if s_list[ele[0]][ele[1]] == '*':
            key = f'{ele[0]}-{ele[1]}'
            num = int(s_list[index_i][index_j_start: index_j_end+1])
            res[key] = num
            return res
    return res


def get_all_sum(s_list):
    """获取所有的数字"""
    length_row = len(s_list)
    length_column = len(s_list[0])
    star_dict = {}
    all_sum = 0
    for i in range(length_row):
        start_j = None
        end_j = None
        for j in range(length_column):
            if s_list[i][j] in '0123456789' and start_j is None:
                start_j = j
                end_j = j
            elif s_list[i][j] in '0123456789' and start_j is not None:
                end_j = j
            elif s_list[i][j] not in '0123456789' and start_j is not None and end_j is not None:
                star_index_num = get_num_or_zero(s_list, length_row, length_column, i, start_j, end_j)
                if star_index_num:
                    key = list(star_index_num.keys())[0]
                    if key in star_dict:
                        star_dict[key].append(star_index_num[key])
                    else:
                        star_dict[key] = [star_index_num[key]]
                start_j = None
                end_j = None

        # 最后一个数字也要处理
        if start_j is not None and end_j is not None:
            star_index_num = get_num_or_zero(s_list, length_row, length_column, i, start_j, end_j)
            if star_index_num:
                key = list(star_index_num.keys())[0]
                if key in star_dict:
                    star_dict[key].append(star_index_num[key])
                else:
                    star_dict[key] = [star_index_num[key]]

    for key in star_dict:
        if len(star_dict[key]) == 2:
            all_sum += star_dict[key][0] * star_dict[key][1]

    return all_sum
