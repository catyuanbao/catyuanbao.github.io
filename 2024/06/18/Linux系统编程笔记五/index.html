背景

主要包括：

  1. Demand paging
  2. Locking memory in RAM
  3. Memory protection

Demand paging

Demand-paging（按需分页） 是一种内存管理技术，在现代操作系统中用于实现虚拟内存。

它的基本原理是将进程所需的内存页（page）按需加载到物理内存中，而不是一次性加载所有页。

这种技术的主要目的是提高内存使用效率和系统性能。

在按需分页系统中，程序开始执行时，只有代码段和栈的一部分被加载到内存。

随着程序执行，对数据段和堆的访问会导致页面故障，将相应的页加载到内存。

比如如下代码：

// 假设我们有一个简单的进程虚拟地址空间
int main() {
    int *data = malloc(1024 * sizeof(int)); // 分配堆内存
    for (int i = 0; i < 1024; i++) {
        data[i] = i; // 当访问这些内存页时，会触发页面故障
    }
    free(data); // 释放内存
    return 0;
}

当 data[i] = i 被执行时，如果对应的页还没有在物理内存中，会发生页面故障。操作系统会：

  1. 查找页面表，确认页不在内存中。
  2. 从磁盘读取该页，将其加载到物理内存。
  3. 更新页面表项，指向物理内存的新位置。
  4. 继续执行被中断的指令。

如果申请的内存块很大，系统可能不会真的把大块内存直接给你，而是当实际使用的时候再去加载。

Resident

Resident 在计算机术语中通常指的是常驻内存的进程或页面。这意味着某些数据或程序部分一直保留在物理内存中，而不被交换到磁盘。

Resident Program：通常指一些关键的系统程序或常用的应用程序，这些程序被设计为常驻内存，以便快速响应用户请求。例如，操作系统内核、设备驱动程序和一些常用的后台服务。

Resident Pages：指在操作系统的虚拟内存管理中，被加载到物理内存且不易被置换出内存的页面。通常，这些页面是频繁访问的页面或关键数据，操作系统会优先保持这些页面常驻内存。

在 Demand-paging 系统中，内存页面按需加载，这意味着只有在访问这些页面时，才会将它们从磁盘加载到内存。

而 Resident 页面的概念则与此相对，指的是那些被标记为不可交换出内存的页面。这样做可以避免频繁的页面置换，提高性能和响应速度。

mincore 系统调用用于确定在指定内存区域中的哪些页面驻留在物理内存中。以下是一个简单的使用 mincore 的示例：

#include <stdio.h>
#include <stdlib.h>
#include <sys/mman.h>
#include <unistd.h>
#include <fcntl.h>
#include <errno.h>
#include <string.h>

int main() {
    size_t page_size = sysconf(_SC_PAGESIZE); // 获取系统页面大小
    size_t length = 4 * page_size; // 分配4个页面大小的内存

    // 分配匿名内存，内存初始化为0
    void *addr = mmap(NULL, length, PROT_READ | PROT_WRITE, MAP_ANONYMOUS | MAP_PRIVATE, -1, 0)
;
    if (addr == MAP_FAILED) {
        perror("mmap");
        return 1;
    }

    // 初始化内存中的一些数据
    memset(addr, 1, length);

    // 分配一个数组，用于存储 mincore 的结果
    unsigned char *vec = (unsigned char *)malloc(length / page_size);
    if (vec == NULL) {
        perror("malloc");
        munmap(addr, length);
        return 1;
    }

    // 调用 mincore
    if (mincore(addr, length, vec) != 0) {
        perror("mincore");
        free(vec);
        munmap(addr, length);
        return 1;
    }

    // 打印结果
    for (size_t i = 0; i < length / page_size; i++) {
        printf("Page %zu is %s\n", i, (vec[i] & 1) ? "in memory" : "not in memory");
    }

    // 释放资源
    free(vec);
    munmap(addr, length);

    return 0;
}

Locking memory

内存锁定是指将某些内存区域固定在物理内存中，防止其被交换到交换空间。

这对于性能敏感的应用程序或需要确保数据安全（如加密密钥）的情况非常重要。

内存锁定可以防止操作系统在内存不足时将这些数据换出到磁盘，从而保证数据始终驻留在内存中。

使用mlock方法，可以锁定部分内存：

#include <stdio.h>
#include <stdlib.h>
#include <sys/mman.h>
#include <string.h>

int main() {
    const size_t size = 4096;  // 要锁定的内存大小（4KB）
    void *buffer;

    // 分配内存
    buffer = malloc(size);
    if (buffer == NULL) {
        perror("malloc");
        exit(EXIT_FAILURE);
    }

    // 锁定内存
    if (mlock(buffer, size) != 0) {
        perror("mlock");
        free(buffer);
        exit(EXIT_FAILURE);
    }

    // 使用内存
    strcpy(buffer, "这是一些敏感数据");
    printf("数据: %s\n", (char *)buffer);

    // 解锁内存
    if (munlock(buffer, size) != 0) {
        perror("munlock");
        free(buffer);
        exit(EXIT_FAILURE);
    }

    // 释放内存
    free(buffer);
    return 0;
}

Limits and privileges

对程序的限制，也有这么一条：

$ prlimit | grep MEMLOCK
MEMLOCK    max locked-in-memory address space 67108864 67108864 字节

最大锁内存的大小。

下面的实例给出如何锁内存：

#define _GNU_SOURCE
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/mman.h>		// for mlock(),...
#include <string.h>
#include <limits.h>
#include "../common.h"

#define CMD_MAX  256

static void disp_locked_mem(void)
{
    char *cmd = malloc(CMD_MAX);
    if (!cmd)
        FATAL("malloc(%zu) failed\n", CMD_MAX);
    snprintf(cmd, CMD_MAX - 1, "grep Lck /proc/%d/status", getpid());
    system(cmd);
    free(cmd);
}

static void try_mlock(const char *cpgs)
{
    size_t num_pg = atol(cpgs);
    const long pgsz = sysconf(_SC_PAGESIZE);
    void *ptr = NULL;
    size_t len;

    len = num_pg * pgsz;
    if (len >= LONG_MAX)
        FATAL("too many bytes to alloc (%zu), aborting now\n", len);

    /* ptr = malloc(len); */
    /* Don't use the malloc; POSIX wants page-aligned memory for mlock */
    posix_memalign(&ptr, pgsz, len);
    if (!ptr)
        FATAL("posix_memalign(for %zu bytes) failed\n", len);

    /* Lock the memory region! */
    if (mlock(ptr, len)) {
        free(ptr);
        FATAL("mlock failed\n");
    }
    printf("Locked %zu bytes from address %p\n", len, ptr);
    memset(ptr, 'L', len);
    disp_locked_mem();
    sleep(1);

    /* Now unlock it.. */
    if (munlock(ptr, len)) {
        free(ptr);
        FATAL("munlock failed\n");
    }
    printf("unlocked..\n");
    free(ptr);
}

int main(int argc, char **argv)
{
    if (argc < 2) {
        fprintf(stderr, "Usage: %s pages-to-alloc\n", argv[0]);
        exit(EXIT_FAILURE);
    }
    disp_locked_mem();
    try_mlock(argv[1]);
    exit(EXIT_SUCCESS);
}

运行结果：

$ ./mlock_try 2000000
VmLck:	       0 kB
FATAL:mlock_try.c:try_mlock:63: posix_memalign(for 8192000000 bytes) failed
  kernel says: Cannot allocate memory

Locking all pages

mlockall 函数用于锁定整个进程地址空间中的所有页面，防止它们被换出到交换空间。

这对于需要确保整个进程数据常驻内存的情况非常有用。

使用实例：

#include <stdio.h>
#include <stdlib.h>
#include <sys/mman.h>
#include <string.h>
#include <unistd.h>

int main() {
    // 锁定当前和将来的所有内存映射
    if (mlockall(MCL_CURRENT | MCL_FUTURE) != 0) {
        perror("mlockall");
        exit(EXIT_FAILURE);
    }

    // 分配并使用一些内存
    const size_t size = 4096;  // 4KB
    void *buffer = malloc(size);
    if (buffer == NULL) {
        perror("malloc");
        exit(EXIT_FAILURE);
    }

    strcpy(buffer, "这是一些敏感数据");
    printf("数据: %s\n", (char *)buffer);

    // 使用一段时间后解除锁定
    sleep(10);

    // 释放内存
    free(buffer);

    // 解锁所有内存映射
    if (munlockall() != 0) {
        perror("munlockall");
        exit(EXIT_FAILURE);
    }

    return 0;
}

Memory protection

mprotect 函数用于设置指定内存区域的保护属性，例如读、写和执行权限。它可以用来改变内存页面的访问权限，从而提高内存访问的安全性。

实例如下：

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/mman.h>

int main() {
    const size_t size = 4096;  // 一个页面的大小

    // 分配一块内存
    void *buffer = mmap(NULL, size, PROT_READ | PROT_WRITE, MAP_PRIVATE | MAP_ANONYMOUS, -1, 0)
;
    if (buffer == MAP_FAILED) {
        perror("mmap");
        exit(EXIT_FAILURE);
    }

    // 写入数据
    sprintf((char *)buffer, "Hello, World!");

    // 打印数据
    printf("数据: %s\n", (char *)buffer);

    // 更改内存保护属性为只读
    if (mprotect(buffer, size, PROT_READ) != 0) {
        perror("mprotect");
        munmap(buffer, size);
        exit(EXIT_FAILURE);
    }

    // 尝试写入只读内存（这将导致段错误）
    // sprintf((char *)buffer, "This will cause a segmentation fault!");

    // 解除内存映射
    if (munmap(buffer, size) != 0) {
        perror("munmap");
        exit(EXIT_FAILURE);
    }

    return 0;
}

An Aside – LSM logs, Ftrace

LSM (Log-Structured Merge-tree) 日志是一种用于存储引擎的数据结构和算法，特别适用于高写入性能需求的应用程序，如日志记录系统、NoSQL数据库等。

LSM 树通过将所有写入操作记录到磁盘上的顺序文件（称为 SSTables 或者 HFiles）来优化写性能，并定期合并这些文件来保持读取效率。

Ftrace 是 Linux 内核中一个用于跟踪和调试内核以及用户空间程序的强大工具。

它可以帮助开发人员和系统管理员了解系统行为，分析性能问题，调试内核模块和驱动程序等。Ftrace 提供了许多功能，包括函数跟踪、时间测量、事件记录等。

Allocation with the alloca(3)

alloca 是一个 C 标准库函数，用于在栈上动态分配内存空间。

它类似于 malloc，但是分配的内存空间是在当前函数的栈帧上，而不是堆上。

因此，内存分配的生命周期与函数调用的生命周期相关联，函数返回时自动释放分配的内存空间，无需显式调用 free。

使用实例：

#include <stdio.h>
#include <alloca.h>

int main() {
    size_t size = 100;  // 要分配的字节数
    char *ptr;

    ptr = (char *)alloca(size);  // 分配内存空间

    if (ptr != NULL) {
        printf("Memory allocated successfully using alloca!\n");
        // 可以在此处使用 ptr 指向的内存空间
    } else {
        printf("Memory allocation failed!\n");
    }

    // 注意：在函数返回时，ptr 指向的内存自动释放，无需手动调用 free

    return 0;
}

栈空间限制：由于分配在栈上，栈空间是有限的，通常比堆空间小得多。因此，不应该分配过大的内存空间，以免发生栈溢出。

可移植性：alloca 函数不是标准 C 函数，可能不受所有编译器的支持。在非 POSIX 环境下可能需要考虑替代方案。

安全性：应该小心使用 alloca，特别是在需要动态分配大量内存或者递归深度大的情况下，以避免不必要的风险。

alloca：适用于需要动态分配临时内存空间且内存需求大小不确定的情况。例如，根据运行时的输入参数动态分配内存空间。
