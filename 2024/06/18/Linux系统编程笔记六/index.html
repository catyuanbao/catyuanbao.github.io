背景

本文主要记录《Hands-On System Programming with Linux》一书中第五章的内容，主要是内存相关的bug问题。

Uninitialized memory access

case 1

未初始化的内存访问，可能运行结果不同，我这里测试的初始化是0

#include <stdio.h>

static void uninit_var() {
    int x;
    if (x)
        printf("true case: x=%d\n", x);
    else
        printf("false case x=%d\n", x);
}

int main() {
    uninit_var();
    return 0;
}

这里x没有初始化，运行结果：

$ ./bug0
false case x=0

$ ./bug0
false case x=0

Out-of-bounds memory accesses

case2

越界写数据，一般在数组中的bug

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(int argc, char** argv) {
    int i;
    int arr[5];
    int temp[8];
    for (i = 0; i <= 500009; i++) {
        arr[i] = 100000;
    }
    return 0;

}

这里越界的最大值不同，运行结果也不同，较小的值不会报错，大些的值运行如下：

$ ./bug1
[1]    1005148 segmentation fault (core dumped)  ./bug1

case3

溢出写入动态内存

#include <stdio.h>
#include <stdlib.h>
#include <string.h>


static void write_overflow(void) {
    char *dest, src[] = "abcdefg1249999";
    dest = malloc(8);

    strcpy(dest, src);
    free(dest);
}

int main() {
    write_overflow();
    return 0;
}

写入数据溢出。

运行并没有报错。

修复代码如下：

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

static void write_overflow(void) {
    char *dest, src[] = "abcdefg1249999";
    dest = malloc(strlen(src) + 1); // Allocate enough memory for the source string + null terminator

    if (dest == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        exit(1);
    }

    strcpy(dest, src);
    printf("Copied string: %s\n", dest);
    free(dest);
}

int main() {
    write_overflow();
    return 0;
}

case4

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    char *p = malloc(8);
    if (p == NULL) {
        return 1;
    }
    p--;
    strncpy(p, "abcd5678", 8);
    free(++p);
    return 0;
}

将指针 p 向后移动一个位置，使其指向分配内存之前的一个字节。这是非常危险的操作，因为它使指针指向未分配的内存区域。

运行失败:

double free or corruption (out)
[1]    1005907 IOT instruction (core dumped)  ./bug4

strncpy 将会拷贝 8 个字符，但是因为 p 指向未分配的内存区域，所以会导致内存越界访问。

修复代码如下：

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    char *p = malloc(9); // 分配足够的内存来存储字符串和终止符
    if (p == NULL) {
        return 1;
    }
    strncpy(p, "abcd5678", 8);
    p[8] = '\0'; // 确保字符串以 null 终止
    printf("Copied string: %s\n", p);
    free(p);
    return 0;
}

case5

读越界

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

static void read_overflow(void) {
    char arr[5], tmp[8];

    memset(arr, 'a', 5);
    memset(tmp, 't', 8);
    tmp[7] = '\0';
    printf("arr = %s\n", arr);
}

int main() {
    read_overflow();
}

代码中存在缓冲区读取溢出的问题

问题的根因：

arr 被填充了 5 个 'a'，但它并没有以 null 终止符结尾。

printf("arr = %s\n", arr) 期望 arr 是一个以 null 终止的字符串。

由于 arr 没有以 null 终止，printf 会继续读取 arr 之后的内存，直到遇到一个 null 字符。这会导致未定义行为，可能会读取到 tmp 或其他内存内容。

修复代码如下：

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

static void read_overflow(void) {
    char arr[6], tmp[8]; // arr 增加一个字节，用于存储 null 终止符

    memset(arr, 'a', 5);
    arr[5] = '\0'; // 设置 null 终止符
    memset(tmp, 't', 8);
    tmp[7] = '\0';

    printf("arr = %s\n", arr); // 现在可以安全地打印 arr
}

int main() {
    read_overflow();
}

case6

在动态内存上读溢出。

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

static void read_overflow_dy(void) {
    char *arr;
    arr = malloc(5);
    if (arr == NULL) {
        return;
    }
    memset(arr, 'a', 5);


    arr[5] = 'S';
    arr[6] = 'e';
    arr[7] = 'c';
    arr[8] = 'r';
    arr[9] = 'e';
    arr[10] = 'T';
    printf("arr = %s\n", arr);

    printf("*(arr+100)=%d\n", *(arr+100));
    printf("*(arr+10000)=%d\n", *(arr+10000));

    free(arr);
}

int main() {
    read_overflow_dy();
}

代码存在多个内存访问越界的问题。

arr 只分配了 5 字节内存，但随后却写入了更多的字符，导致内存越界。

arr[5] 到 arr[10] 的写操作都是越界访问。

*(arr+100) 和 *(arr+10000) 是对未分配内存的访问，导致未定义行为，可能会导致程序崩溃或其他不可预料的行为。

修复代码：

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

static void read_overflow_dy(void) {
    char *arr;
    arr = malloc(11); // 分配足够的内存来存储所有字符和一个null终止符
    if (arr == NULL) {
        return;
    }
    memset(arr, 'a', 5);

    arr[5] = 'S';
    arr[6] = 'e';
    arr[7] = 'c';
    arr[8] = 'r';
    arr[9] = 'e';
    arr[10] = 'T';
    arr[11] = '\0'; // 添加null终止符
    printf("arr = %s\n", arr);

    // 不应该访问未分配的内存
    // printf("*(arr+100)=%d\n", *(arr+100));
    // printf("*(arr+10000)=%d\n", *(arr+10000));

    free(arr);
}

int main() {
    read_overflow_dy();
}

case7

内存读取下溢（Read Underflow）是指尝试读取分配内存区域之前的数据。

这种情况可能会导致未定义行为，因为读取的内存可能未被分配或被用于其他目的，可能会导致程序崩溃或返回意外值。

代码如下：

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

static void read_underflow(void) {
    char *arr;
    arr = malloc(5);
    if (arr == NULL) {
        return;
    }
    memset(arr, 'a', 5);

    // 假设在数组起始之前访问数据
    char value_before = *(arr - 1);
    printf("Value before the allocated memory: %d\n", value_before);

    free(arr);
}

int main() {
    read_underflow();
}

运行结果:

$ ./bug7
Value before the allocated memory: 0

修复代码：

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

static void safe_memory_access(void) {
    char *arr;
    arr = malloc(5);
    if (arr == NULL) {
        return;
    }
    memset(arr, 'a', 5);

    // 正常访问分配的内存，不进行下溢访问
    printf("arr = %.*s\n", 5, arr); // 打印前 5 个字符

    free(arr);
}

int main() {
    safe_memory_access();
}

Use-after-free/Use-after-return bug

case8

“Use After Free”（UAF）是一种常见的内存错误，通常发生在程序中试图使用已经被释放的内存。

这种错误可能导致程序崩溃、不可预测的行为，甚至可能被攻击者利用来执行任意代码，进而导致安全漏洞。

双重释放（Double Free）：同一块内存被多次释放，导致释放后的内存仍被访问。

悬挂指针（Dangling Pointer）：指向已释放内存的指针没有被设置为NULL，导致其被再次访问。

错误的内存管理：由于复杂的内存管理或多线程环境下的竞态条件，导致内存释放后仍然被访问。

实例代码：

#include <stdio.h>
#include <stdlib.h>

void use_after_free() {
    char *ptr = malloc(10);
    if (ptr == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        return;
    }
    // 使用内存
    snprintf(ptr, 10, "Hello");

    // 释放内存
    free(ptr);

    // 释放后继续使用内存
    printf("%s\n", ptr);  // 未定义行为
}

int main() {
    use_after_free();
    return 0;
}

运行结果：

$ ./bug8
J
$ ./bug8
X

是否后不应该继续使用内存。

case9

“Use After Return”（UAR）是一种内存错误，类似于“Use After Free”（UAF），但发生在函数返回后。

具体来说，UAR错误发生在程序试图使用一个函数返回后其栈帧已经被释放的局部变量或内存。

返回局部变量的指针：在函数返回时，返回局部变量的指针，导致指针指向已经被销毁的栈帧。

返回局部分配的内存：返回函数内部分配的内存，函数返回后该内存被销毁，指针指向无效内存。

使用局部变量的引用：函数返回后继续使用局部变量的引用，导致引用指向无效内存。

实例：

#include <stdio.h>

char* dangerous_function() {
    char buffer[10];
    snprintf(buffer, 10, "Hello");
    return buffer;  // 返回局部变量的指针
}

int main() {
    char *str = dangerous_function();
    printf("%s\n", str);  // 未定义行为，可能导致崩溃
    return 0;
}

运行结果：

$ ./a.out
[1]    1006470 segmentation fault (core dumped)  ./a.out

避免返回局部变量的指针

使用动态内存分配

使用全局变量或静态变量

使用返回值传递数据

case10

“双重释放”（Double-Free）是一种内存管理错误，发生在程序试图释放同一块内存两次或多次。

这种错误可能导致程序崩溃、未定义行为，甚至可能被攻击者利用来执行恶意代码，进而引发严重的安全漏洞。

代码实例：

#include <stdio.h>
#include <stdlib.h>

void double_free() {
    char *ptr = malloc(10);
    if (ptr == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        return;
    }

    // 正常使用内存
    snprintf(ptr, 10, "Hello");

    // 第一次释放内存
    free(ptr);

    // 第二次释放同一块内存
    free(ptr);  // 未定义行为，可能导致崩溃
}

int main() {
    double_free();
    return 0;
}

运行结果：

$ ./bug10
free(): double free detected in tcache 2
[1]    1006535 IOT instruction (core dumped)  ./bug10

解决方法：

  1. 释放内存后立即将指针设置为NULL，以避免再次释放。
  2. 使用智能指针（在C++中）
  3. 内存检查工具

    char *ptr;
    ptr = malloc(1024*1024);
    memset(ptr, 0, 1024*1024);

    memleaky();
