import time

def read_data(s):
    """读取数字到列表"""
    s_list = []
    for line in s.strip().split('\n'):
        s_list.append(line)
    return s_list


def get_num_or_zero(s_list, length_row, length_column, index_i, index_j_start, index_j_end):
    """获取数值或者0，不满足条件返回0，否则返回数字"""

    other_ele = [(index_i, index_j_start-1), (index_i, index_j_end+1)]

    for j in range(index_j_start-1, index_j_end+2):
        other_ele.append((index_i-1, j))
        other_ele.append((index_i+1, j))

    ret = []

    for ele in other_ele:
        if ele[0] < 0 or ele[1] >= length_column or ele[1] < 0 or ele[0] >= length_row:
            continue
        else:
            ret.append(ele)
    for ele in ret:
        if s_list[ele[0]][ele[1]] != '.':
            return int(s_list[index_i][index_j_start: index_j_end+1])
    return 0


def get_all_sum(s_list):
    """获取所有的数字"""
    length_row = len(s_list)
    length_column = len(s_list[0])
    all_sum = 0
    for i in range(length_row):
        line_count = 0
        line_sum = 0
        start_j = None
        end_j = None
        for j in range(length_column):
            if s_list[i][j] in '0123456789' and start_j is None:
                start_j = j
                end_j = j
            elif s_list[i][j] in '0123456789' and start_j is not None:
                end_j = j
            elif s_list[i][j] not in '0123456789' and start_j is not None and end_j is not None:
                num = get_num_or_zero(s_list, length_row, length_column, i, start_j, end_j)
                line_sum += num
                all_sum += num
                # print(f'line {i}, num {num}')
                line_count += 1
                start_j = None
                end_j = None

        # 最后一个数字也要处理
        if start_j is not None and end_j is not None:
            num = get_num_or_zero(s_list, length_row, length_column, i, start_j, end_j)
            line_sum += num
            all_sum += num
            # print(f'line {i}, num {num}')
            line_count += 1
        # print(f'line {i}, count {line_count}, sum {line_sum}')
    # print(f'Total sum: {all_sum}')
    return all_sum

import random

# 定义文件名
filename = "in1.txt"

# 行数和列数
num_rows = 12000
num_cols = 12000

# 根据示例计算各字符的比例
example_text = """
467..114..
...*......
..35..633.
......#...
617*......
.....+.58.
..592.....
......755.
...$.*....
.664.598..
"""

# 将示例文本展开成一个字符串并计算各字符的比例
char_counts = {}
total_chars = 0

for char in example_text:
    if char != '\n':
        if char in char_counts:
            char_counts[char] += 1
        else:
            char_counts[char] = 1
        total_chars += 1

# 计算每个字符的概率
char_probabilities = {char: count / total_chars for char, count in char_counts.items()}

# 创建字符集和对应的概率列表
char_set = list(char_probabilities.keys())
probabilities = list(char_probabilities.values())

# 生成文本文件
with open(filename, "w") as file:
    for _ in range(num_rows):
        line = ''.join(random.choices(char_set, probabilities, k=num_cols))
        file.write(line + '\n')

print(f"生成的文本已写入{filename}文件")

    FILE * file;

线程优化版本：

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <pthread.h>
#include <time.h>

#define MAX_LINES 20000
#define LINE_LENGTH 20000

// 定义计算数字的索引
struct NumPos {
    int line_index;     // 行索引
    int column_start;   // 列开始索引
    int column_end;     // 列结束索引
};

// 定义线程参数结构
struct ThreadData {
    char **array;
    struct NumPos *num_pos;
    int start_idx;
    int end_idx;
    long long partial_sum;
    int thread_num;
    int lines_count;
};


// 读取文件到数组
void read_lines(char** array, int max_lines, int *lines_count) {
    FILE * file;
    char filename[] = "in1.txt";
    // 打开文件并按行读取
    file = fopen(filename, "r");
    if (file == NULL) {
        perror("打开文件失败");
        return;
    }

    int i = 0;
    while (i < max_lines && fgets(array[i], LINE_LENGTH, file)) {
        size_t len = strlen(array[i]);
        // 读到行末，停止
        if (len > 0 && array[i][len - 1] == '\n') {
            array[i][len - 1] = '\0';
        }
        i++;
    }
    *lines_count = i;
    // 关闭文件
    fclose(file);
}


// 计算出可能的数据索引信息，为后续计算数字准备
void compute_num_pos(char** array, struct NumPos* a, int max_lines, int *len_num_pos) {

long long compute_num(char** text, int max_lines, struct NumPos* num_pos, int index) {
    int line_index = num_pos[index].line_index;
    int c_start = num_pos[index].column_start;
    int c_end = num_pos[index].column_end;

    int how_many_round = ((c_end - c_start) + 1 + 2) * 3 - (c_end - c_start + 1);
    int array[100][2] = {-1};

    array[0][0] = line_index;
    array[0][1] = c_start - 1;

    array[1][0] = line_index;
    array[1][1] = c_end + 1;

void* compute_thread_func(void* arg) {
    struct ThreadData* args = (struct ThreadData*)arg;
    int i;
    // 每个线程计算自己的部分
    for (i = args->start_idx; i < args->end_idx; i++) {
        args->partial_sum += compute_num(args->array, args->lines_count,
                                         args->num_pos, i);
    }
    return (void*)0;
}


int main(int argc, char* argv[]) {
    if (argc != 2) {
        fprintf(stderr, "Usage: %s thread_num\n", argv[0]);
        return 1;
    }

    long THREAD_NUM = strtol(argv[1], NULL, 10);

    // 需要使用动态内存，否则会报错，栈内存溢出
    char** array = (char**)calloc(MAX_LINES, sizeof(char*));
    for (int i = 0; i < MAX_LINES; i++) {
        array[i] = (char*)calloc(LINE_LENGTH, sizeof(char));
    }

    int lines_count = 0;
    read_lines(array, MAX_LINES, &lines_count);

    // 初始化num_pos内存
    struct NumPos* num_pos = NULL;
    num_pos = malloc(sizeof(struct NumPos) * (MAX_LINES * MAX_LINES));
    if (num_pos == NULL) {
        printf("malloc error\n");
        return 1;
    }

    int len_num_pos;
    compute_num_pos(array, num_pos, lines_count, &len_num_pos);

    struct timespec start, end;
    clock_gettime(CLOCK_MONOTONIC, &start);

    // 初始化线程
    struct ThreadData* thread_args;
    pthread_t* thread_ids;
    thread_ids = malloc(sizeof(pthread_t) * THREAD_NUM);
    if (thread_ids == NULL) {
        printf("malloc error!\n");
        return 2;
    }

    thread_args = malloc(sizeof(struct ThreadData) * THREAD_NUM);
    if (thread_args == NULL) {
        printf("malloc error!\n");
        return 3;
    }

    int i;
    int chunk_size = (int)(len_num_pos / THREAD_NUM);

    for (i = 0; i < THREAD_NUM; i++) {
        thread_args[i].array = array;
        thread_args[i].num_pos = num_pos;
        thread_args[i].start_idx = i * chunk_size;
        thread_args[i].end_idx = i * chunk_size + chunk_size;
        if (i == THREAD_NUM - 1) {
            thread_args[i].end_idx = len_num_pos;
        }
        thread_args[i].thread_num = i;
        thread_args[i].lines_count = lines_count;
        thread_args[i].partial_sum = 0;
        pthread_create(&thread_ids[i], NULL, compute_thread_func, &thread_args[i]);
    }

    for (i = 0; i < THREAD_NUM; i++) {
        pthread_join(thread_ids[i], NULL);
    }

    long long sum = 0;
    for (i = 0; i < THREAD_NUM; i++) {
        sum += thread_args[i].partial_sum;
    }

    clock_gettime(CLOCK_MONOTONIC, &end);
    long seconds = end.tv_sec - start.tv_sec;
    long nanoseconds = end.tv_nsec - start.tv_nsec;
    double elapsed = seconds + nanoseconds*1e-9;

    printf("Time elapsed: %.9f seconds\n", elapsed);

    // 释放内存
    for (i = 0; i < MAX_LINES; i++) {
        free(array[i]);
    }
    free(array);
    free(num_pos);
    free(thread_ids);
    free(thread_args);

    printf("%lld\n", sum);
    return 0;
}
