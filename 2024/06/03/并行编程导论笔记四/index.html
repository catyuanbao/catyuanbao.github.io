背景

本文主要记录《An Introduction to Parallel Programming 》第二版 一书 第三章 Distributed memory programming with MPI。

这部分主要是学习使用MPI编程接口，做简单的多进程并行计算应用程序开发。

本文主要通过一些练习，进一步学习MPI编程。

同步-按顺序打印数据

每个进程按照rank_id打印自己的数据，学习如何同步多个进程：

#include <mpi.h>
#include <stdio.h>

int main(int argc, char* argv[]) {
    MPI_Init(&argc, &argv);

    int rank, size;
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    // 每个进程都运行整个循环，但是都只打印和自己rank一致的数据
    for (int i = 0; i < size; ++i) {
        if (i == rank) {
            fprintf(stdout, "Greeting from %d\n", i);
            fflush(stdout);
        }
        MPI_Barrier(MPI_COMM_WORLD); // 等待所有进程都执行完本次循环
    }

    MPI_Finalize();
    return 0;
}

两个进程交替打印100内奇数偶数

经典面试问题，两个线程交替打印奇数偶数，这里改成MPI实现：

#include <mpi.h>
#include <stdio.h>

int main(int argc, char* argv[]) {
    MPI_Init(&argc, &argv);

    int rank, size;
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    if (size != 2) {
        if (rank == 0) {
            fprintf(stderr, "This program requires exactly 2 processes.\n");
        }
        MPI_Finalize();
        return 1;
    }

    int number = (rank == 0) ? 1 : 2; // Rank 0 打印奇数，Rank 1 打印偶数
    int max_num = 100; // 每个进程打印的次数

    for (int i = 0; i < max_num; ++i) {
        if ((i % 2 == 0 && rank == 0) || (i % 2 != 0 && rank == 1)) {
            printf("Rank %d: %d\n", rank, number);
            fflush(stdout); // 刷新输出缓冲区
            number += 2;
        }
        MPI_Barrier(MPI_COMM_WORLD); // 等待所有进程完成一次打印
    }

    MPI_Finalize();
    return 0;
}

Write an MPI program that computes a tree-structured global sum. First write your program for the special case in which comm_sz is a power of two. Then, after you’ve gotten this version working, modify your program so that it can handle any comm_sz.

在并行计算中，树状结构（tree-structured）的全局求和是一种高效的方式来计算所有进程数据的总和。这种方法利用树状结构的特性，通过分阶段的通信逐步汇总数据，从而减少通信次数和延迟。

2的整数次幂的解法：

#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char** argv) {
    MPI_Init(&argc, &argv);

    int comm_sz;  // Number of processes
    int my_rank;  // Process rank

    MPI_Comm_size(MPI_COMM_WORLD, &comm_sz);
    MPI_Comm_rank(MPI_COMM_WORLD, &my_rank);

    // Initialize local value
    int local_value = my_rank + 1;

    int partner;
    int sum = local_value;

    // Ensure comm_sz is a power of two for simplicity
    if ((comm_sz & (comm_sz - 1)) != 0) {
        if (my_rank == 0) {
            printf("Number of processes must be a power of two.\n");
        }
        MPI_Finalize();
        return 1;
    }

    for (int stage = 1; stage < comm_sz; stage <<= 1) {
        // 匹配交换数据的对象
        partner = my_rank ^ stage;
        int received_value;
                // 发送并接受数据
        MPI_Sendrecv(&sum, 1, MPI_INT, partner, 0, &received_value, 1, MPI_INT, partner, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
        sum += received_value;
    }

    if (my_rank == 0) {
        printf("Total sum: %d\n", sum);
    }

    MPI_Finalize();
    return 0;
}

处理任意个数的进程情况

为了处理任意数量的进程数（不一定是 2 的幂），需要对不满足条件的进程进行特殊处理。

一个常见的策略是：

  • 找到最大的 2 的幂（称为 p2) 小于或等于 comm_sz。
  • 额外的进程将其值发送到较低编号的进程，然后退出。
  • 剩余的 p2 个进程使用上述树状汇总算法进行计算。
#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char** argv) {
    MPI_Init(&argc, &argv);

    int comm_sz;
    int my_rank;

    MPI_Comm_size(MPI_COMM_WORLD, &comm_sz);
    MPI_Comm_rank(MPI_COMM_WORLD, &my_rank);

    int p2 = 1;
    while (p2 <= comm_sz) {
        p2 <<= 1;
    }
    p2 >>= 1;  // Largest power of 2 less than or equal to comm_sz

    int local_value = my_rank + 1;
    int sum = local_value;

    if (my_rank >= p2) {
        MPI_Send(&local_value, 1, MPI_INT, my_rank - p2, 0, MPI_COMM_WORLD);
    } else {
        if (my_rank + p2 < comm_sz) {
            int received_value;
            MPI_Recv(&received_value, 1, MPI_INT, my_rank + p2, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
            sum += received_value;
        }

        for (int stage = 1; stage < p2; stage <<= 1) {
            int partner = my_rank ^ stage;
            int received_value;
            MPI_Sendrecv(&sum, 1, MPI_INT, partner, 0, &received_value, 1, MPI_INT, partner, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
            sum += received_value;
        }

        if (my_rank == 0) {
            printf("Total sum: %d\n", sum);
        }
    }

    MPI_Finalize();
    return 0;
}
