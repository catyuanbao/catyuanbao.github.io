橘猫元宝

基本概念

  1. 进程是资源管理的最小单位，线程是程序执行的最小单位；

  2. 每个进程有自己的数据段，代码段，堆栈段。线程通常叫做轻量级进程，包括独立的栈，CPU寄存器状态

    线程是进程的一条执行路径，每个线程共享进程的所有资源，包括open fd,内存等；

  3. 线程和进程比起来很小，花费更少的资源；

  4. 当进程启动时，默认会有一个线程，称为主控线程，可以再派生多个子线程，单个线程被分配到某个CPU执行；

  5. 线程分为用户级线程和内核级线程，一般情况用户级线程和内核级线程一一绑定，跟随系统调度分配策略运行；

thread_self

每个线程都有自己的ID；

线程ID只在所属的进程中有效。

The pthread_self() function returns the thread ID of the calling thread.


#include <pthread.h>
#include <stdio.h>

int main() {
    pthread_t main_thread_num;
    main_thread_num = pthread_self();
    printf("main thread id is %lld\n", (long long)main_thread_num);
    return 0;
}

两个线程实现龟兔赛跑：


#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <unistd.h>


void* th_fn(void* arg) {

    int distance = (int)arg;
    int i;
    for (i = 0; i <= distance; i++) {
        printf("%lx run %d\n", (unsigned long)pthread_self(), i);
        sleep((double)0.01);
    }

    return (void*)0;
}


int main() {
    int err;
    pthread_t rabbit, turtle;
    int distance = 50;
    if ((err = pthread_create(&rabbit, NULL,
                    th_fn, (void*)distance)) != 0) {
        perror("pthread create error!");
    }

    if ((err = pthread_create(&turtle, NULL,
                    th_fn, (void*)distance)) != 0) {
        perror("pthread create error!");
    }

    pthread_join(rabbit, NULL);
    pthread_join(turtle, NULL);
  
    printf("main thread is %lx\n", (unsigned long)pthread_self());
    printf("end\n");

    return 0;
}

#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    char name[20];
    int number;
    int time;
    int start;
    int end;
    int step;
}RaceArg;


void* th_fn(void* arg) {
        // 每个线程有单独的栈空间，并不共享这个函数的变量
    RaceArg *racer = (RaceArg*)arg;
    int i;
    int count = 0;
    for (i = racer->start; i <= racer->end; i += racer->step) {
        printf("%s run %d distance\n", racer->name, i);
        sleep(racer->time);
        // 兔子中间睡了120秒
        if (racer->number == 0 && count == 20) {
            sleep(120);
            // 醒来跑得更快了
            racer->step = 3;
        }
        count++;
    }

    printf("%s run %d distance\n", racer->name, i);
    return (void*)0;
}


int main() {
    int err;
    pthread_t rabbit, turtle;

    RaceArg rabbit_arg = {"rabbit", 0, 1, 1, 100, 2};
    RaceArg turtle_arg = {"turtle", 1, 2, 1, 100, 1};

                    th_fn, (void*)&rabbit_arg)) != 0) {
        perror("pthread create error!");
    }

    if ((err = pthread_create(&turtle, NULL,
                    th_fn, (void*)&turtle_arg)) != 0) {
        perror("pthread create error!");
    }

    pthread_join(rabbit, NULL);

    printf("end\n");

修改龟兔赛跑的代码，将每个线程运行的时间通过pthread_join返回


#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <unistd.h>
#include <time.h>


typedef struct {
    char name[20];
    int number;
    int time;
    int start;
    int end;
    int step;
} RaceArg;


void* th_fn(void* arg) {

    RaceArg *racer = (RaceArg*)arg;
    int i;
    int count = 0;

    time_t current_time, end_time;

    // 获取当前时间戳
    current_time = time(NULL);

    for (i = racer->start; i <= racer->end; i += racer->step) {
        printf("%s run %d distance\n", racer->name, i);
        sleep(racer->time);
        // 兔子中间睡了120秒
        if (racer->number == 0 && count == 20) {
            sleep(120);
            // 醒来跑得更快了
            racer->step = 3;
        }
        count++;
    }

    end_time = time(NULL);

    printf("%s run %d distance\n", racer->name, i);
    // 计算时间差，并将其转换为 void* 类型返回
    return (void*)(intptr_t)(end_time - current_time);
}


int main() {
    int err;
    pthread_t rabbit, turtle;

    RaceArg rabbit_arg = {"rabbit", 0, 1, 1, 100, 2};
    RaceArg turtle_arg = {"turtle", 1, 2, 1, 100, 1};

    if ((err = pthread_create(&rabbit, NULL,
                    th_fn, (void*)&rabbit_arg)) != 0) {
        perror("pthread create error!");
    }

    if ((err = pthread_create(&turtle, NULL,
                    th_fn, (void*)&turtle_arg)) != 0) {
        perror("pthread create error!");
    }

    // 用于存储线程函数返回值的变量，类型为 void*
    void* cost_time_r;
    void* cost_time_t;

    pthread_join(rabbit, &cost_time_r);
    pthread_join(turtle, &cost_time_t);
    printf("main thread is %lx\n", (unsigned long)pthread_self());

    // 将返回的 void* 类型的指针转换回 int 类型，并打印输出
    printf("rabbit cost time is %d\n", (int)(intptr_t)cost_time_r);
    printf("turtle cost time is %d\n", (int)(intptr_t)cost_time_t);

    return 0;
}

void* th_func(void *arg) {
    int execute = (int)arg;
    pthread_cleanup_push(clean_up_func, "first call");
    pthread_cleanup_push(clean_up_func, "second call");

    pthread_t t1, t2;

    pthread_create(&t1, NULL, th_func, (void*)1);
    pthread_join(t1, NULL);
    pthread_create(&t2, NULL, th_func, (void*)0);

    pthread_join(t2, NULL);

    return 0;
}

以分离状态启动的线程，可以在不调用pthread_join的时候，自动回收资源。(一般在网络通信中的线程使用)

实例:


#include <pthread.h>
#include <stdlib.h>
#include <stdio.h>
#include <unistd.h>
#include <string.h>


void out_state(pthread_attr_t *attr) {
    int state;

    if (pthread_attr_getdetachstate(attr, &state) != 0) {
        perror("get detach state error");
    } else {
        // PTHREAD_CREATE_DETACHED;
        if (state == PTHREAD_CREATE_JOINABLE) {
            printf("joinable state\n");
        } else if (state == PTHREAD_CREATE_DETACHED) {
            printf("detached state\n");
        } else {
            printf("error state\n");
        }
    }
}


void* th_func(void *arg) {
    int i;
    int sum = 0;
    for (i = 1; i <= 100; i++) {
        sum += i;
    }

    return (void*)sum;
}


int main() {
    int err;
    pthread_t default_th, detach_th;

    pthread_attr_t attr;
    pthread_attr_init(&attr);

    out_state(&attr);

    pthread_create(&default_th, &attr, th_func, (void*)0);
    int res;
    if ((err = pthread_join(default_th, (void*)&res)) != 0 ) {
        perror("pthread join error!\n");
    } else {
        printf("defualt return is %d\n", (int)res);
    }

    pthread_attr_destroy(&attr);

    printf("********************\n");

    pthread_attr_setdetachstate(&attr,
            PTHREAD_CREATE_DETACHED);
    out_state(&attr);

    // 以分离状态启动线程
    pthread_create(&detach_th, &attr,
            th_func, (void*)0);

    if ((err = pthread_join(detach_th, (void*)&res)) != 0 ) {
        perror("pthread join error!\n");
        fprintf(stderr, "join error for %s", strerror(err));
    } else {
        printf("defualt return is %d\n",(int)res);
    }

    pthread_attr_destroy(&attr);

    sleep(1);

    return 0;
}

使用互斥锁（pthread_mutex_t）解决对共享资源访问。

多个线程想要对已经上锁互斥资源访问，会被挂起，直到拿到释放的资源。

#include <pthread.h>

int pthread_mutex_init(pthread_mutex_t *mutex, const pthread_mutexattr_t *attr);
int pthread_mutex_destroy(pthread_mutex_t *mutex);

int pthread_mutex_lock(pthread_mutex_t *mutex); // 拿不到锁会阻塞
int pthread_mutex_unlock(pthread_mutex_t *mutex); 
int pthread_mutex_trylock(pthread_mutex_t *mutex); // 拿不到锁不阻塞，会返回错误信息

互斥锁实例:


#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>
#include <assert.h>
#include <string.h>

typedef struct {
    int code;
    double balance;

    // mutex 对账号加锁操作
    // 和绑定账号在一起,尽量不使用全局变量.
    pthread_mutex_t mutex;
}Account;


Account* create_account(int code, double balance);
void destroy_account(Account *a);
double withdraw(Account *a, double amt);
double deposit(Account *a, double amt);
double get_balance(Account *a);

Account* create_account(int code, double balance) {
    Account *a = (Account*)malloc(sizeof(Account));
    assert(a != NULL);
    a->code = code;
    a->balance = balance;
    // 对锁初始化
    pthread_mutex_init(&a->mutex, NULL);
    return a;
}

void destroy_account(Account *a) {
    assert(a != NULL);
    free(a);
    // 销毁锁
    pthread_mutex_destroy(&a->mutex);
}

double withdraw(Account *a, double amt) {
    assert(a != NULL);
    // 加锁操作
    pthread_mutex_lock(&a->mutex);
    if (amt < 0 || amt > a->balance) {
        pthread_mutex_unlock(&a->mutex);
        return 0.0;
    }
    double balance = a->balance;
    sleep(1);
    balance -= amt;
    a->balance = balance;
    pthread_mutex_unlock(&a->mutex);
    return amt; 
}

double deposit(Account *a, double amt) {
    assert(a != NULL);
    pthread_mutex_lock(&a->mutex);
    if (amt < 0) {
        pthread_mutex_unlock(&a->mutex);
        return 0.0;
    }
    double balance = a->balance;
    sleep(1);
    balance += amt;
    a->balance = balance;
    pthread_mutex_unlock(&a->mutex);
    return amt;  
}

double get_balance(Account *a) {
    assert(a != NULL);
    double balance = a->balance;
    return balance;
}

                                        int pshared);

// 设置互斥锁类型
// 1.标准互斥锁（阻塞式的） 2.递归互斥锁（同一个线程可以多次获取锁） 3.检错互斥锁
int pthread_mutexattr_gettype(const pthread_mutexattr_t *restrict attr,
           int *restrict type);
int pthread_mutexattr_settype(pthread_mutexattr_t *attr, int type);


#include <pthread.h>
#include <stdlib.h>
#include <stdio.h>
#include <string.h>

int main(int argc, char *argv[]) {
    pthread_mutex_t mutex;
    if (argc < 2) {
        printf("-usage %s [error|normal|recursive]", argv[0]);
        exit(1);
    }

    // 互斥锁属性
    pthread_mutexattr_t attr;
    pthread_mutexattr_init(&attr);

    if (!strcmp(argv[1], "normal")) {
       pthread_mutexattr_settype(&attr, 
        PTHREAD_MUTEX_NORMAL); 
    } else if(!strcmp(argv[1], "error")) {
       pthread_mutexattr_settype(&attr, 
        PTHREAD_MUTEX_ERRORCHECK); 
    } else if (!strcmp(argv[1], "recursive")) {
       pthread_mutexattr_settype(&attr, 
        PTHREAD_MUTEX_RECURSIVE); 
    }

    // 初始化互斥锁
    pthread_mutex_init(&mutex, &attr);

    if (pthread_mutex_lock(&mutex) != 0) {
        printf("lock failure!\n");
    } else {
        printf("lock success\n");
    }

    if (pthread_mutex_lock(&mutex) != 0) {
        printf("lock failure!\n");
    } else {
        printf("lock success\n");
    }

    pthread_mutex_unlock(&mutex);
    pthread_mutex_unlock(&mutex);

    pthread_mutexattr_destroy(&attr);
    pthread_mutex_destroy(&mutex);

    return 0;
}

递归锁Python使用实例：

import threading

# 创建递归互斥锁对象
recursive_lock = threading.RLock()

def recursive_function(count):
    with recursive_lock:
        print("Acquired lock, count:", count)
        if count > 0:
            recursive_function(count - 1)
        print("Released lock, count:", count)

# 测试递归函数

读写锁

线程使用互斥锁，读并发性不高，当读多写少时，可以使用读写锁提高读并发性。

读写锁类型：pthread_rwlock_t


#include <pthread.h>

int pthread_rwlock_rdlock(pthread_rwlock_t *rwlock);  // 加读锁
int pthread_rwlock_tryrdlock(pthread_rwlock_t *rwlock); // 
int pthread_rwlock_wrlock(pthread_rwlock_t *rwlock); // 加写锁
int pthread_rwlock_unlock(pthread_rwlock_t *rwlock); // 释放锁

读写锁实例：

#include <pthread.h>
#include <stdio.h>

#include <string.h>


// 定义读写锁
pthread_rwlock_t rwlock;

int main(int argc, char *argv[]) {
    if (argc < 3) {
        printf("-usage: %s [r|w] [r|w]]\n", argv[0]);
        return 1;
    }
    
    // 读写锁初始化
    pthread_rwlock_init(&rwlock, NULL);

    if (!strcmp("r", argv[1])) {
        // 加读锁
        if (pthread_rwlock_rdlock(&rwlock) != 0) {
            printf("first read lock failure\n");
        } else {
            printf("first read lock success\n");
        }
    } else if (!strcmp("w", argv[1])) { 
        if (pthread_rwlock_wrlock(&rwlock) != 0) {
            printf("first write lock failure\n");
        } else {
            printf("first write lock success\n");
        }
    }

    if (!strcmp("r", argv[2])) {
        // 加读锁
        if (pthread_rwlock_rdlock(&rwlock) != 0) {
            printf("second read lock failure\n");
        } else {
            printf("second read lock success\n");
        }
    } else if (!strcmp("w", argv[2])) { 
        if (pthread_rwlock_wrlock(&rwlock) != 0) {
            printf("second write lock failure\n");
        } else {
            printf("second write lock success\n");
        }
    }

    pthread_rwlock_unlock(&rwlock);
    pthread_rwlock_unlock(&rwlock);

    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>
#include <assert.h>
#include <string.h>

typedef struct {
    int code;
    double balance;

    // mutex 对账号加锁操作
    // 和绑定账号在一起,尽量不使用全局变量.
    pthread_rwlock_t rwlock;
}Account;


Account* create_account(int code, double balance);
void destroy_account(Account *a);
double withdraw(Account *a, double amt);
double deposit(Account *a, double amt);
double get_balance(Account *a);

Account* create_account(int code, double balance) {
    Account *a = (Account*)malloc(sizeof(Account));
    assert(a != NULL);
    a->code = code;
    a->balance = balance;
    // 对锁初始化
    // pthread_mutex_init(&a->mutex, NULL);
    pthread_rwlock_init(&a->rwlock, NULL);
    return a;
}

void destroy_account(Account *a) {
    assert(a != NULL);
    free(a);
    // 销毁锁
    pthread_rwlock_destroy(&a->rwlock);
}

double withdraw(Account *a, double amt) {
    assert(a != NULL);
    // 加写锁操作
    pthread_rwlock_wrlock(&a->rwlock);

    if (amt < 0 || amt > a->balance) {
        pthread_rwlock_unlock(&a->rwlock);
        return 0.0;
    }
    double balance = a->balance;
    sleep(1);
    balance -= amt;
    a->balance = balance;
    pthread_rwlock_unlock(&a->rwlock);
    return amt; 
}

double deposit(Account *a, double amt) {
    assert(a != NULL);
    pthread_rwlock_wrlock(&a->rwlock);
    if (amt < 0) {
        pthread_rwlock_unlock(&a->rwlock);
        return 0.0;
    }
    double balance = a->balance;
    sleep(1);
    balance += amt;
    a->balance = balance;
    pthread_rwlock_unlock(&a->rwlock);
    return amt;  
}

double get_balance(Account *a) {
    assert(a != NULL);
    pthread_rwlock_rdlock(&a->rwlock);
    double balance = a->balance;
    pthread_rwlock_unlock(&a->rwlock);
    return balance;
}
