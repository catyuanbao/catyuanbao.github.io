背景

本文主要记录LeetCode 876 删除链表节点过程，需要注意删除的为头结点时候的判断。

实现

C语言描述：

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

typedef struct Node {
    int val;
    struct Node* next;
} ListNode;

// 分配内存并检查是否成功
ListNode* allocate_node(int value) {
    ListNode* node = malloc(sizeof(ListNode));
    if (node == NULL) {
        fprintf(stderr, "malloc error in line %d\n", __LINE__);
        exit(1);
    }
    node->val = value;
    node->next = NULL;
    return node;
}

// 构建链表，包含1到end_nums的数字
ListNode* build_list_by_nums(int end_nums) {
    if (end_nums < 1) {
        fprintf(stderr, "arg end_nums error %d should be more than 1\n", end_nums);
        exit(1);
    }

    ListNode* head = allocate_node(1);

    for (int i = 2; i <= end_nums; i++) {
        ListNode* cur = allocate_node(i);
        cur->next = head;
        head = cur;
    }
    return head;
}

// 显示链表
void display_list(ListNode* head) {
    ListNode* cur = head;
    while (cur) {
        fprintf(stdout, "node at %p with value %d, next node at %p\n", cur, cur->val, cur->next);
        cur = cur->next;
    }
}

// 释放链表
void free_list(ListNode* head) {
    ListNode* cur = head;
    while (cur) {
        ListNode* next = cur->next;
        free(cur);
        fprintf(stdout, "free node at %p\n", cur);
        cur = next;
    }
}

// 删除链表中指定元素值的节点，返回头节点
ListNode* delete_by_value(ListNode* head, int value) {
    // 判断是否头结点
    if (head == NULL) {
        return head;
    }
    if (head->val == value) {
        head = head->next;
    } else {
        ListNode* pre = head;
        // 找到前一个节点
        while (pre != NULL && pre->next != NULL && pre->next->val != value) {
            pre = pre->next;
        }
        pre->next = pre->next->next;
    }

    return head;

}

int main() {
    ListNode* head = build_list_by_nums(10);
    display_list(head);
    head = delete_by_value(head, 10);
    display_list(head);

    head = delete_by_value(head, 8);
    display_list(head);

    free_list(head);
    return 0;
}

class Node(object):
    def __init__(self, val=0, next_node=None):
        self.val = val
        self.next_node = next_node


class LinkList(object):

    def __init__(self, end_nums):
        if end_nums < 0:
            raise ValueError("end_nums should bigger than 0")

        self.head = None
        self.end_nums = end_nums


    def make_list(self):
        self.head = Node(val=1)

        for i in range(2, self.end_nums + 1):
            cur_node = Node(val=i, next_node=self.head)
            self.head = cur_node


    def display(self):
        cur = self.head
        while cur is not None:
            print(f"node at {cur} with val {cur.val} next node is {cur.next_node}")
            cur = cur.next_node


    def delete_by_value(self, val):
        if self.head is None:
            return

        if self.head.val == val:
            self.head = self.head.next_node
        else:
            pre_node = self.head
            while pre_node is not None \
                    and pre_node.next_node is not None \
                    and pre_node.next_node.val != val:
                pre_node = pre_node.next_node

            pre_node.next_node = pre_node.next_node.next_node


if __name__ == '__main__':
    link_list = LinkList(10)
    link_list.make_list()
    link_list.display()

    link_list.delete_by_value(8)
    link_list.display()

 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };

struct ListNode* deleteNode(struct ListNode* head, int val){
    if (head == NULL) {
        return head;
    }
    if (head->val == val) {
        head = head->next;
    } else {
        struct ListNode* pre = head;
        while (pre != NULL && pre->next != NULL && pre->next->val != val) {
            pre = pre->next;
        }
        pre->next = pre->next->next;
    }
    return head;

}
