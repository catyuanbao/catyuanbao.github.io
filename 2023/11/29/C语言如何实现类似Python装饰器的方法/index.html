通过函数指针，把函数当做变量传递给统计函数

#include <stdio.h>
#include <time.h>

// 定义一个装饰器函数，用于统计函数执行时间
void timing_decorator(void (*function)()) {
    clock_t start_time, end_time;

    // 记录函数开始时间
    start_time = clock();

    // 调用被装饰的函数
    function();

    // 记录函数结束时间
    end_time = clock();

    // 计算并输出执行时间
    double elapsed_time = ((double) (end_time - start_time)) / CLOCKS_PER_SEC;
    printf("Elapsed time1: %f seconds\n", elapsed_time);
}

// 被装饰的函数1
int example_function1() {
    // 模拟一些工作
    for (int i = 0; i < 100000000; i++) {
        // do nothing
    }
    return 10;
}

// 被装饰的函数2
void example_function2() {
    // 模拟一些工作
    for (int i = 0; i < 50000000; i++) {
        // do nothing
    }
}

int main() {
    // 使用装饰器统计函数执行时间
    timing_decorator(example_function1);
    timing_decorator(example_function2);

    return 0;
}

运行效果：

Elapsed time1: 0.143700 seconds
Elapsed time1: 0.071566 seconds

如何传递参数？

使用结构体，把参数从结构体透传进函数：


#include <stdio.h>
#include <time.h>

// 定义一个结构体, 用于传递参数
struct ArgsStruct {
    int data1;
    int data2;
};

// 定义一个接受带有结构体参数的函数指针类型
typedef void (*FunctionWithStructParam)(struct ArgsStruct);

// 定义一个装饰器函数，用于统计函数执行时间
void timing_decorator(FunctionWithStructParam function, struct ArgsStruct param) {
    clock_t start_time, end_time;

    // 记录函数开始时间
    start_time = clock();

    // 调用被装饰的函数
    function(param);

    // 记录函数结束时间
    end_time = clock();

    // 计算并输出执行时间
    double elapsed_time = ((double)(end_time - start_time)) / CLOCKS_PER_SEC;
    printf("Elapsed time: %f seconds\n", elapsed_time);
}

// 被装饰的带有结构体参数的函数
void example_function_with_struct(struct ArgsStruct exampleStruct) {
    // 模拟一些工作
    for (int i = 0; i < exampleStruct.data1; i++) {
        // do nothing
    }
    for (int j = 0; j < exampleStruct.data2; j++) {
        // do nothing
    }
}

int main() {
    // 使用装饰器统计带有结构体参数的函数执行时间
    struct ArgsStruct param1 = {100000000, 0};
    struct ArgsStruct param2 = {50000000, 0};

    timing_decorator(example_function_with_struct, param1);
    timing_decorator(example_function_with_struct, param2);

    return 0;
}

运行实例：

Elapsed time: 0.135690 seconds
Elapsed time: 0.068217 seconds

如何把结果保存下来？

还是使用结构体，把结果保存下来。


#include <stdio.h>
#include <time.h>

// 定义一个结构体, 用于传递参数
struct ArgsStruct {
    int data1;
    int data2;
    int result;
};

// 定义一个接受带有结构体指针参数的函数指针类型
typedef void (*FunctionWithStructParam)(struct ArgsStruct* args);

// 定义一个装饰器函数，用于统计函数执行时间
void timing_decorator(FunctionWithStructParam function, struct ArgsStruct* param) {
    clock_t start_time, end_time;

    // 记录函数开始时间
    start_time = clock();

    // 调用被装饰的函数
    function(param);

    // 记录函数结束时间
    end_time = clock();

    // 计算并输出执行时间
    double elapsed_time = ((double)(end_time - start_time)) / CLOCKS_PER_SEC;
    printf("Elapsed time: %f seconds\n", elapsed_time);
}

// 被装饰的带有结构体参数的函数
void example_function_with_struct(struct ArgsStruct *exampleStruct) {
    // 模拟一些工作
    for (int i = 0; i < exampleStruct->data1; i++) {
        // do nothing
    }
    for (int j = 0; j < exampleStruct->data2; j++) {
        // do nothing
    }
    exampleStruct->result = exampleStruct->data1 + exampleStruct->data2;
}

int main() {
    // 使用装饰器统计带有结构体参数的函数执行时间
    struct ArgsStruct param1 = {100000000, 1};
    struct ArgsStruct param2 = {50000000, 1};

    timing_decorator(example_function_with_struct, &param1);
    timing_decorator(example_function_with_struct, &param2);

    printf("result1 %d\n", param1.result);
    printf("result2 %d\n", param2.result);


    return 0;
}

运行结果：

Elapsed time: 0.151823 seconds
Elapsed time: 0.075114 seconds
result1 100000001
result2 50000001

如何把运行时间传递出来？

还是通过结构体，把运行耗时保存在结构体中：

#include <stdio.h>
#include <time.h>

// 定义一个结构体, 用于传递参数
struct ArgsStruct {
    int data1;
    int data2;
    int result;
    float cost_time;
};

// 定义一个接受带有结构体指针参数的函数指针类型
typedef void (*FunctionWithStructParam)(struct ArgsStruct* args);

// 定义一个装饰器函数，用于统计函数执行时间
void timing_decorator(FunctionWithStructParam function, struct ArgsStruct* param) {
    clock_t start_time, end_time;

    // 记录函数开始时间
    start_time = clock();

    // 调用被装饰的函数
    function(param);

    // 记录函数结束时间
    end_time = clock();

    // 计算并输出执行时间
    double elapsed_time = ((double)(end_time - start_time)) / CLOCKS_PER_SEC;
    printf("Elapsed time: %f seconds\n", elapsed_time);
    param->cost_time = elapsed_time;
}

// 被装饰的带有结构体参数的函数
void example_function_with_struct(struct ArgsStruct *exampleStruct) {
    // 模拟一些工作
    for (int i = 0; i < exampleStruct->data1; i++) {
        // do nothing
    }
    for (int j = 0; j < exampleStruct->data2; j++) {
        // do nothing
    }
    exampleStruct->result = exampleStruct->data1 + exampleStruct->data2;
}

int main() {
    // 使用装饰器统计带有结构体参数的函数执行时间
    struct ArgsStruct param1 = {100000000, 1};
    struct ArgsStruct param2 = {50000000, 1};

    timing_decorator(example_function_with_struct, &param1);
    timing_decorator(example_function_with_struct, &param2);

    printf("result1 %d cost_time: %f\n", param1.result, param1.cost_time);
    printf("result2 %d cost_time: %f\n", param2.result, param2.cost_time);


    return 0;
}

运行实例：

Elapsed time: 0.128669 seconds
Elapsed time: 0.064927 seconds
result1 100000001 cost_time: 0.128669
result2 50000001 cost_time: 0.064927
